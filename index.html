contract Token {
    storage {
        total_supply: u64,
        balances: Map
    }

    public fn transfer(to: Address, amount: u64) -> Result<()> {
        let sender = msg::sender();
        ensure!(self.balances[sender] >= amount);
        
        self.balances[sender] -= amount;
        self.balances[to] += amount;
        
        emit Transfer(sender, to, amount);
        Ok(())
    }
}
